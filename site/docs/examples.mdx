### Confetti clicking

```js
let siteColors = ['#ffa68d', '#fd3a84'];
document.getElementById('buttonConfetti').addEventListener('click', function (e) {
    e.preventDefault();
    party.element(this, {
        color: siteColors,
        count: party.variation(25, 0.5),
        size: party.minmax(6, 10),
        velocity: party.minmax(-300, -600),
        angularVelocity: party.minmax(6, 9)
    });
});
```

### Making it rain

```js
let siteColors = ['#ffa68d', '#fd3a84'];
document.getElementById('screenConfetti').addEventListener('click', function (e) {
    e.preventDefault();
    party.screen({
        color: siteColors,
        size: party.minmax(6, 12),
        count: party.variation(300 * (window.innerWidth / 1980), 0.4),
        angle: -180,
        spread: 80,
        angularVelocity: party.minmax(6, 9)
    });
});
```

### Additional shapes

```js
let siteColors = ['#ffa68d', '#fd3a84'];
document.getElementById('shapeConfetti').addEventListener('click', function (e) {
    e.preventDefault();
    party.element(this, {
        shape: ['star', 'heart'],
        count: 20,
        color: siteColors,
        spread: 360,
        gravity: false,
        size: party.minmax(14, 22),
        velocity: party.minmax(-300, -600),
        angularVelocity: party.minmax(2, 4)
    });
});
```
